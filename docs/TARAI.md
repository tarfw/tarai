// workers/agents/conversation-agent.ts
import { VoltAgent, Tool, Memory } from '@voltagent/voltagent';

export class ConversationAgent extends VoltAgent {
  name = 'ConversationAgent';
  description = 'Main agent that handles user conversations and routes to specialists';

  // Memory stored in Redis
  memory: Memory;

  // Available tools
  tools: Tool[] = [
    {
      name: 'vector_search',
      description: 'Search marketplace using semantic vectors',
      parameters: {
        type: 'object',
        properties: {
          query: { type: 'string' },
          commerce_type: { type: 'string', enum: ['product', 'service', 'booking', 'event', 'all'] },
          filters: { type: 'object' },
        },
      },
      handler: async (params: any) => {
        return await this.vectorSearch(params);
      },
    },
    {
      name: 'get_semantic_suggestions',
      description: 'Get semantic suggestions for autocomplete',
      parameters: {
        type: 'object',
        properties: {
          partial_query: { type: 'string' },
          context: { type: 'string' },
        },
      },
      handler: async (params: any) => {
        return await this.getSemanticSuggestions(params);
      },
    },
    {
      name: 'check_availability',
      description: 'Check availability for booking/service',
      parameters: {
        type: 'object',
        properties: {
          listing_id: { type: 'string' },
          date: { type: 'string' },
          time: { type: 'string' },
        },
      },
      handler: async (params: any) => {
        return await this.checkAvailability(params);
      },
    },
    {
      name: 'create_task',
      description: 'Create task for user (buyer or seller)',
      parameters: {
        type: 'object',
        properties: {
          user_id: { type: 'string' },
          task_type: { type: 'string' },
          task_data: { type: 'object' },
        },
      },
      handler: async (params: any) => {
        return await this.createTask(params);
      },
    },
    {
      name: 'process_booking',
      description: 'Create booking and assign tasks',
      parameters: {
        type: 'object',
        properties: {
          listing_id: { type: 'string' },
          buyer_id: { type: 'string' },
          seller_id: { type: 'string' },
          booking_details: { type: 'object' },
        },
      },
      handler: async (params: any) => {
        return await this.processBooking(params);
      },
    },
  ];

  async chat(message: string, userId: string, conversationId: string) {
    // Load conversation memory from Redis
    const context = await this.memory.load(conversationId);

    // Understand user intent
    const intent = await this.analyzeIntent(message, context);

    // Execute appropriate workflow
    switch (intent.type) {
      case 'search':
        return await this.handleSearch(message, intent, userId);
      case 'booking':
        return await this.handleBooking(message, intent, userId, conversationId);
      case 'purchase':
        return await this.handlePurchase(message, intent, userId);
      case 'question':
        return await this.handleQuestion(message, intent);
      default:
        return await this.handleGeneral(message);
    }
  }

  private async handleBooking(
    message: string,
    intent: any,
    userId: string,
    conversationId: string
  ) {
    const context = await this.memory.load(conversationId);

    // Multi-step booking flow
    if (!context.booking_type) {
      // Step 1: Get booking type from semantic suggestions
      const suggestions = await this.tools[1].handler({
        partial_query: message,
        context: 'booking',
      });

      await this.memory.save(conversationId, {
        step: 'select_type',
        suggestions,
      });

      return {
        type: 'suggestions',
        message: 'What would you like to book?',
        suggestions: suggestions, // ['ğŸš• Taxi', 'ğŸ¬ Movie', 'ğŸ¨ Hotel', ...]
      };
    }

    if (!context.from_location) {
      // Step 2: Get "from" location
      return {
        type: 'input',
        message: 'Where are you starting from?',
        input_type: 'location',
        placeholder: 'Enter pickup location',
      };
    }

    if (!context.to_location) {
      // Step 3: Get "to" location
      return {
        type: 'input',
        message: 'Where are you going?',
        input_type: 'location',
        placeholder: 'Enter destination',
      };
    }

    // Step 4: Search available taxis using vectors
    const available = await this.tools[0].handler({
      query: `${context.booking_type} from ${context.from_location} to ${context.to_location}`,
      commerce_type: 'service',
      filters: {
        location: context.from_location,
        available: true,
      },
    });

    if (available.length === 0) {
      return {
        type: 'message',
        message: 'Sorry, no drivers available right now. Try again in a few minutes?',
      };
    }

    // Step 5: Show available options as cards
    return {
      type: 'selection',
      message: `Found ${available.length} available drivers:`,
      options: available.map(driver => ({
        id: driver.id,
        title: driver.title,
        subtitle: `â‚¹${driver.price} â€¢ ${driver.rating}â­ â€¢ ${driver.distance}km away`,
        image: driver.thumbnail_url,
        eta: driver.eta_minutes,
      })),
      action: 'confirm_booking',
    };
  }

  private async processBooking(params: any) {
    const { listing_id, buyer_id, seller_id, booking_details } = params;

    // Create order in InstantDB
    const orderId = await this.createOrder({
      listing_id,
      buyer_id,
      seller_id,
      ...booking_details,
    });

    // Create tasks for both parties
    await Promise.all([
      // Buyer tasks
      this.createTask({
        user_id: buyer_id,
        task_type: 'await_driver',
        task_data: {
          order_id: orderId,
          title: 'Waiting for driver to accept',
          status: 'pending',
        },
      }),

      // Seller (driver) tasks
      this.createTask({
        user_id: seller_id,
        task_type: 'new_booking',
        task_data: {
          order_id: orderId,
          title: 'New taxi booking',
          actions: ['Accept', 'Decline'],
          status: 'pending',
        },
      }),
    ]);

    // Send notifications
    await Promise.all([
      this.sendPushNotification(buyer_id, 'Booking created! Waiting for driver confirmation'),
      this.sendPushNotification(seller_id, 'New taxi booking request!'),
    ]);

    return {
      type: 'confirmation',
      message: 'âœ… Booking created! Driver will confirm shortly.',
      order_id: orderId,
      next_steps: [
        'Driver will accept booking',
        'You will receive notification',
        'Track driver in real-time',
      ],
    };
  }

  private async vectorSearch(params: any) {
    // Generate embedding (if not provided)
    const embedding = params.embedding ||
      await this.generateEmbedding(params.query);

    // Query Turso with vector
    const results = await this.queryTurso({
      sql: `
        SELECT *,
               vector_distance_cos(embedding, ?) as similarity
        FROM listings
        WHERE status = 'active'
          AND type = ?
          AND vector_distance_cos(embedding, ?) < 0.3
        ORDER BY similarity ASC
        LIMIT 20
      `,
      args: [embedding, params.commerce_type, embedding],
    });

    return results;
  }

  private async getSemanticSuggestions(params: any) {
    const { partial_query, context } = params;

    // Generate embedding for partial query
    const embedding = await this.generateEmbedding(partial_query);

    // Find similar categories/types
    const similar = await this.queryTurso({
      sql: `
        SELECT DISTINCT category, type, COUNT(*) as count
        FROM listings
        WHERE vector_distance_cos(embedding, ?) < 0.4
        GROUP BY category, type
        ORDER BY count DESC
        LIMIT 10
      `,
      args: [embedding],
    });

    // Return as suggestions
    return similar.map(s => ({
      text: s.category,
      type: s.type,
      icon: this.getIconForType(s.type),
      count: s.count,
    }));
  }
}

// app/(tabs)/ai.tsx

export default function AITab() {
  const [messages, setMessages] = useState<Message[]>([]);
  const [input, setInput] = useState('');
  const [suggestions, setSuggestions] = useState<string[]>([]);

  const sendMessage = async (text: string) => {
    // Add user message
    setMessages(prev => [...prev, { role: 'user', content: text }]);

    // Call VoltAgent
    const response = await fetch(`${WORKER_URL}/agent/chat`, {
      method: 'POST',
      body: JSON.stringify({
        message: text,
        user_id: user.id,
        conversation_id: conversationId,
      }),
    });

    const agentResponse = await response.json();

    // Handle different response types
    switch (agentResponse.type) {
      case 'suggestions':
        setSuggestions(agentResponse.suggestions);
        setMessages(prev => [...prev, {
          role: 'agent',
          content: agentResponse.message,
          type: 'suggestions',
          suggestions: agentResponse.suggestions,
        }]);
        break;

      case 'input':
        setMessages(prev => [...prev, {
          role: 'agent',
          content: agentResponse.message,
          type: 'input',
          input_type: agentResponse.input_type,
        }]);
        break;

      case 'selection':
        setMessages(prev => [...prev, {
          role: 'agent',
          content: agentResponse.message,
          type: 'selection',
          options: agentResponse.options,
        }]);
        break;

      case 'confirmation':
        setMessages(prev => [...prev, {
          role: 'agent',
          content: agentResponse.message,
          type: 'confirmation',
        }]);
        break;
    }
  };

  return (
    <View className="flex-1">
      {/* Chat Messages */}
      <ScrollView className="flex-1 p-4">
        {messages.map((msg, idx) => (
          <MessageBubble key={idx} message={msg} />
        ))}
      </ScrollView>

      {/* Input */}
      <View className="p-4 border-t">
        <TextInput
          value={input}
          onChangeText={setInput}
          placeholder="What do you need?"
          className="border rounded-lg p-3"
        />
        <TouchableOpacity onPress={() => sendMessage(input)}>
          <Text>Send</Text>
        </TouchableOpacity>
      </View>
    </View>
  );
}

function MessageBubble({ message }: { message: Message }) {
  if (message.type === 'suggestions') {
    return (
      <View>
        <Text>{message.content}</Text>
        <View className="flex-row flex-wrap gap-2 mt-2">
          {message.suggestions.map(s => (
            <TouchableOpacity
              key={s.text}
              className="bg-blue-100 px-4 py-2 rounded-full"
              onPress={() => selectSuggestion(s)}
            >
              <Text>{s.icon} {s.text}</Text>
            </TouchableOpacity>
          ))}
        </View>
      </View>
    );
  }

  if (message.type === 'selection') {
    return (
      <View>
        <Text>{message.content}</Text>
        {message.options.map(option => (
          <DriverCard
            key={option.id}
            driver={option}
            onSelect={() => confirmBooking(option.id)}
          />
        ))}
      </View>
    );
  }

  // Default text message
  return (
    <View className={message.role === 'user' ? 'self-end' : 'self-start'}>
      <Text>{message.content}</Text>
    </View>
  );
}

function TaskCard({ task }: { task: Task }) {
  const { transact } = useInstantDB();

      case 'Accept':
        await acceptBooking(task.order_id);
        await updateTaskStatus(task.id, 'in_progress');
        break;

        await declineBooking(task.order_id);
        await updateTaskStatus(task.id, 'cancelled');

        // Create next task

### VoltAgent Setup

```typescript
// workers/agents/conversation-agent.ts
import { VoltAgent, Tool, Memory } from '@voltagent/voltagent';
import { Redis } from '@upstash/redis/cloudflare';

export class ConversationAgent extends VoltAgent {
  memory: Memory;

  constructor(env: Env) {
    super();

    // Setup memory with Redis
    this.memory = new Memory({
      redis: new Redis({
        url: env.UPSTASH_REDIS_URL,
        token: env.UPSTASH_REDIS_TOKEN,
      }),
    });
  }

  tools = [
    {
      name: 'vector_search',
      description: 'Search marketplace using semantic similarity',
      parameters: {
        type: 'object',
        properties: {
          query: { type: 'string' },
          commerce_type: { type: 'string' },
          filters: { type: 'object' },
        },
      },
      handler: async (params) => await this.vectorSearch(params),
    },
    {
      name: 'create_task',
      description: 'Create task for user',
      parameters: {
        type: 'object',
        properties: {
          user_id: { type: 'string' },
          task_type: { type: 'string' },
          task_data: { type: 'object' },
        },
      },
      handler: async (params) => await this.createTask(params),
    },
  ];

  async chat(message: string, userId: string, conversationId: string) {
    // Load conversation memory
    const context = await this.memory.load(conversationId);

    // Process with VoltAgent's flow engine
    const response = await this.process({
      message,
      userId,
      context,
    });

    // Save updated memory
    await this.memory.save(conversationId, response.context);

    return response;
  }
}
```

      task_data: params.task_data,
